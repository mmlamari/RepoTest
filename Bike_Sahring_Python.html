#import RandomForestRegressor
from sklearn.ensemble import RandomForestRegressor
bigdata = pd.DataFrame()

#loop by year then Month
for year in range(2011,2013):
    print year
    for month in range(1,13):
        print 'Filter Data by month/year'
        dfTrain = dfBikes[(dfBikes['Month'] == month) & (dfBikes['Year'] == year)]
        dfTrain = dfTrain[[ 'holiday', 'workingday', 'weather', 'temp', 'atemp', 'humidity', 'windspeed', 'weekday',
               'hour','count']]
        
               
        X = dfTrain.ix[0:,0:9].values
        Y = dfTrain.ix[0:,9:].values
        y =Y.reshape(len(Y),1)
        X =X.reshape(len(X),9)
        
        print 'Split training data 20% fro testing the alogorithm'
        X_train, X_test, y_train, y_test = cross_validation.train_test_split(X, y, test_size=0.20, random_state=0)
        
        rfr = RandomForestRegressor(n_estimators=400,max_features=9, max_depth=None,
        min_samples_split=1, random_state=0)
        print 'fit Trainng data'
        rfr.fit(X_train, y_train.ravel())
        
        print 'predict test Trainng data'
        pred = rfr.predict(X_test)
        exp = y_test
        
        print("train score: {0:.3f}, test score: {1:.3f}\n".format(
        rfr.score(X_train,y_train), rfr.score(X_test, y_test)))
        print("RandomForestRegressor test Residual sum of squares : %.2f" 
      % np.sqrt(np.mean((rfr.predict(X_test) - y_test) ** 2)))
        
        print 'plot expected vs predicted'
        plt.scatter (exp,pred)
        plt.plot([0, 1200] ,[0,1200] ,'k')
        plt.show()
        
        print 'get My test data for same month and yesr last 10 days'
        dffinal = dfBikestest[(dfBikestest['Month'] == month) & (dfBikestest['Year'] == year)]
        dffinal = dffinal[['datetime', 'holiday', 'workingday', 'weather', 'temp', 'atemp', 'humidity', 'windspeed', 'weekday',
               'hour']]
        X_t = dffinal.ix[0:,1:].values
        X_t =X_t.reshape(len(X_t),9)
        
        print 'Predict count'
        count = rfr.predict(X_t)
        count = count.reshape(len(count),1)
        
        dffinal['count']= np.floor(count)
        dffinal=dffinal[['datetime' ,'count']]
        bigdata=bigdata.append(dffinal,ignore_index = True)


print 'Save test data for submit'
